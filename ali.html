<script>
// 🔥 PUT YOUR FIREBASE CONFIG
const firebaseConfig = {
  apiKey: "YOUR_KEY",
  authDomain: "YOUR_PROJECT.firebaseapp.com",
  projectId: "YOUR_PROJECT"
};

firebase.initializeApp(firebaseConfig);

const ADMIN_EMAIL = "youradmin@gmail.com";

// NAV
function show(id){
  document.querySelectorAll('.section').forEach(s=>s.classList.remove('active'));
  document.getElementById(id).classList.add('active');
}

// AUTH
function signup(){
  firebase.auth().createUserWithEmailAndPassword(email.value, pass.value);
}

function login(){
  firebase.auth().signInWithEmailAndPassword(email.value, pass.value)
  .then(()=>{ show('home'); loadMyCourses(); });
}

// ADMIN CHECK
firebase.auth().onAuthStateChanged(user=>{
  if(user && user.email===ADMIN_EMAIL){
    document.getElementById("admin").style.display="block";
  } else {
    document.getElementById("admin").style.display="none";
  }
});

// NOTES
function addNote(){
  firebase.firestore().collection("notes").add({content:noteText.value});
}

firebase.firestore().collection("notes").onSnapshot(snap=>{
  notes.innerHTML="";
  snap.forEach(doc=>{
    notes.innerHTML += "<div class='card'>"+doc.data().content+"</div>";
  });
});

// COURSES
function addCourse(){
  firebase.firestore().collection("courses").add({
    title: courseTitle.value,
    price: coursePrice.value
  });
}

firebase.firestore().collection("courses").onSnapshot(snap=>{
  courseList.innerHTML="";
  snap.forEach(doc=>{
    let d=doc.data();
    courseList.innerHTML += `
      <div class="card">
        <h3>${d.title}</h3>
        <p>$${d.price}</p>
        <button onclick="buy('${doc.id}')">Buy</button>
      </div>`;
  });
});

// BUY
function buy(id){
  let uid=firebase.auth().currentUser.uid;
  firebase.firestore().collection("users")
  .doc(uid).collection("purchases").doc(id)
  .set({paid:true});
  alert("Purchased!");
}

// DASHBOARD
function loadMyCourses(){
  let uid=firebase.auth().currentUser.uid;
  firebase.firestore()
  .collection("users").doc(uid)
  .collection("purchases")
  .onSnapshot(snap=>{
    myCourses.innerHTML="";
    snap.forEach(doc=>{
      myCourses.innerHTML += "<div class='card'>Course Purchased</div>";
    });
  });
}

// VIDEO
function uploadVideo(file){
  let ref=firebase.storage().ref("videos/"+file.name);
  ref.put(file);
}

// LANGUAGE
function setLang(lang){
  if(lang==="am"){
    document.getElementById("loginText").innerText="ግባ / መመዝገብ";
  } else {
    document.getElementById("loginText").innerText="Login / Signup";
  }
}
